function renderLop(key) {
  const container = document.getElementById("lop" + key);
  if (!container) return;

  const chapters = {};
  QUIZ_PAGES.filter(p => p.lop === key).forEach(p => {
    if (!chapters[p.chapter]) chapters[p.chapter] = [];
    chapters[p.chapter].push(p);
  });

  const colors = CLASS_CHAPTER_COLORS[key] || CHAPTER_COLORS; // fallback nếu chưa khai báo

  let html = "";
  let idx = 0;
  for (const chapter in chapters) {
    const color = colors[idx % colors.length];
    html += `<div class="chapter"><h3 style="background:${color}">${chapter}</h3>`;
    chapters[chapter].forEach(item => {
      html += `
        <div class="quiz-card">
          <a href="${item.file.startsWith('http') ? item.file : baseUrl + '?page=' + item.file}" target="_blank">${item.title}</a>
        </div>
      `;
    });
    html += `</div>`;
    idx++;
  }

  container.innerHTML = html;
}
