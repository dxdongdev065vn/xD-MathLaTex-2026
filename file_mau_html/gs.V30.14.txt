// Hàm chính xử lý yêu cầu POST
function doPost(e) {
  const lock = LockService.getScriptLock();
  lock.waitLock(30000); 

  try {
    const payload = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();

    if (payload.action && payload.action === "submit_key") {
      handleAnswerKeySubmission(ss, payload);
      return createJsonResponse({ status: "success", message: "Answer key updated." });
    }

    return handleStudentSubmission(ss, payload);

  } catch (err) {
    Logger.log(err.toString());
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("KetQua");
    if (sheet) {
      sheet.appendRow([new Date(), "SCRIPT ERROR", err.message, JSON.stringify(e.postData.contents)]);
    }
    return createJsonResponse({ status: "error", message: err.toString() });
  } finally {
    lock.releaseLock();
  }
}

// Xử lý bài nộp của học sinh
function handleStudentSubmission(ss, payload) {
  const sheetName = payload.sheetName || "KetQua";
  const sheet = ss.getSheetByName(sheetName) || ss.insertSheet(sheetName);
  const { results, questionTypes } = payload;
  if (!results || !questionTypes || results.length === 0) {
      throw new Error("Dữ liệu nộp bài không hợp lệ, thiếu kết quả hoặc loại câu hỏi.");
  }

  if (sheet.getLastRow() === 0) {
    createInitialHeader(sheet, results.length);
  }

  writeStudentData(sheet, payload);
  const resultRowIndex = sheet.getLastRow();
  const formula = createScoreFormula(resultRowIndex, results.length, questionTypes);
  sheet.getRange(resultRowIndex, 5 + results.length).setFormula(formula);
  
  return createJsonResponse({ status: "success", message: "Results received successfully." });
}

/**
 * Tạo header và khu vực cấu hình điểm.
 */
function createInitialHeader(sheet, questionCount) {
  if (questionCount === 0) return;

  // Dòng 1: Tiêu đề chính
  const mainHeader = ["Thời gian nộp", "Họ và tên", "Lớp", "Tên bài kiểm tra", ...Array.from({ length: questionCount }, (_, i) => "Câu " + (i + 1)), "Tổng điểm"];
  sheet.appendRow(mainHeader);
  // Định dạng sau khi thêm
  sheet.getRange(1, 1, 1, mainHeader.length).setFontWeight("bold");

  // Dòng 2: Tiêu đề khu vực tính điểm
  sheet.appendRow(["QUY TẮC TÍNH ĐIỂM (Giáo viên có thể sửa các ô màu vàng)"]);
  // Định dạng sau khi thêm
  sheet.getRange(2, 1, 1, mainHeader.length).mergeAcross().setHorizontalAlignment("center").setFontWeight("bold").setBackground("#cfe2f3");
  
  // Các dòng cấu hình điểm
  sheet.appendRow(["Điểm Loại 1 (Trắc nghiệm):", 0.25]);
  sheet.appendRow(["Điểm Loại 2 (0 -> 4+ ý đúng):", 0, 0.1, 0.25, 0.5, 1]);
  sheet.appendRow(["Điểm Loại 3 (Điền đáp số):", 1]);
  sheet.appendRow([""]); 
  
  // Định dạng các ô còn lại
  sheet.getRange("A3:A5").setFontWeight("bold");
  sheet.getRange("B3").setBackground("#fff2cc");
  sheet.getRange("B4:F4").setBackground("#fff2cc");
  sheet.getRange("B5").setBackground("#fff2cc");
  sheet.setFrozenRows(5);
}

// Ghi 2 dòng dữ liệu của học sinh
function writeStudentData(sheet, payload) {
  const { name, class: className, quizName, answers, results } = payload;
  const timestamp = new Date();
  const answerRowData = [timestamp, name, className, quizName, ...answers];
  const resultRowData = ["", "", "", "-> Kết quả:", ...results];
  sheet.appendRow(answerRowData);
  sheet.appendRow(resultRowData);
}

// Tạo công thức tính tổng điểm
function createScoreFormula(resultRow, numQuestions, questionTypes) {
  const formulaParts = questionTypes.map((type, i) => {
    const colLetter = String.fromCharCode('A'.charCodeAt(0) + 4 + i);
    const resultCell = colLetter + resultRow;
    switch (type) {
      case 'type1': return `IF(${resultCell}=1; $B$3; 0)`;
      case 'type2': return `IFERROR(CHOOSE(${resultCell} + 1; $B$4; $C$4; $D$4; $E$4; $F$4); 0)`;
      case 'type3': return `IF(${resultCell}=1; $B$5; 0)`;
      default: return "0";
    }
  });
  return "=" + formulaParts.join(" + ");
}

// Tạo phản hồi JSON
function createJsonResponse(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
